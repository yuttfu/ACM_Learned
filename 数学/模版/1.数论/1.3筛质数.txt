#include<iostream>
using namespace std;

const int N = 100010;

int primes[N],cnt;
bool st[N];

void get_prime(int n)
{
    for (int i = 2;i <= n;i++) {
        if (!st[i]) {
            primes[cnt ++] = i;
            for (int j = i+i;j <= n;j += i) st[j] = true;
        }
    }
}//埃氏筛法

void get_prime(int n){
    for(int i = 2;i <= n;i++)
    {
        if(!st[i]) primes[cnt ++] = i;
            for(int j = 0;primes[j] <= n / i;j++)
            {
                st[primes[j]*i] = true;
                if (i % primes[j] == 0) break;
            }
    }
}//线性筛法

int main(){
    int n;

    cin >> n;

    get_prime(n);

    for (int i = 0;i < cnt;i++) printf("%d ",primes[i]);

    puts("");

    return 0;
}
